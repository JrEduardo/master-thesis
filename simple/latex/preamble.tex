\newenvironment{main_abstract_en}[1]
{
  \setlength{\parindent}{0pt}
  \chapter*{\hspace{7.45cm}{ABSTRACT}}
  \addcontentsline{toc}{chapter}{Abstract}
  \begin{center}
    {\bfseries \sffamily \fontsize{12}{12} #1}
    \vspace*{-0.8cm}
    \begin{abstract}
    \parindent0pt
    \noindent
    }{%
    \end{abstract}
  \end{center}
}
